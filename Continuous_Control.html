Replay Buffer Implementation¶

This class implements Experience Replay, a critical component of modern deep reinforcement learning algorithms. The Replay Buffer stores and manages past experiences for more efficient and stable learning.

Key features:

  • Uses a double-ended queue (deque) with fixed maximum size to store experiences
  • Each experience is a tuple containing (state, action, reward, next_state, done)
  • Supports random sampling of experiences for batch learning
  • Handles conversion of experiences to PyTorch tensors and device placement

Main methods:

  • __init__: Initializes the buffer with specified size and batch parameters
  • add: Stores new experiences in the buffer
  • sample: Randomly samples a batch of experiences and converts them to tensors
  • __len__: Returns current number of experiences in buffer

The buffer helps break correlations in sequential data and enables the agent to learn from past experiences multiple times, improving sample efficiency and training stability.

from collections import namedtuple, deque
import random
import numpy as np
import torch
from typing import Tuple, List, Any

class ReplayBuffer:
    """A fixed-size buffer to store experience tuples for deep reinforcement learning.
    
    This implementation provides efficient storage and sampling of experience tuples
    (state, action, reward, next_state, done) for training deep RL agents. It uses
    a double-ended queue with a maximum size to automatically handle buffer overflow.
    
    Attributes:
        action_size (int): Dimension of each action
        memory (deque): Double-ended queue storing experiences
        batch_size (int): Size of each training batch
        experience (namedtuple): Template for storing experience tuples
        device (torch.device): Device for tensor operations
        
    Example:
        >>> buffer = ReplayBuffer(action_size=4, buffer_size=100000, batch_size=64, seed=42)
        >>> state = env.reset()
        >>> action = agent.act(state)
        >>> next_state, reward, done, _ = env.step(action)
        >>> buffer.add(state, action, reward, next_state, done)
    """
    
    def __init__(self, action_size: int, buffer_size: int, batch_size: int, seed: int):
        """Initialize a ReplayBuffer object.
        
        Args:
            action_size (int): Dimension of each action
            buffer_size (int): Maximum size of buffer
            batch_size (int): Size of each training batch
            seed (int): Random seed for reproducibility
        """
        self.action_size = action_size
        self.memory = deque(maxlen=buffer_size)
        self.batch_size = batch_size
        self.experience = namedtuple("Experience", 
                                   field_names=["state", "action", "reward", "next_state", "done"])
        self.seed = random.seed(seed)
        self.device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu")

    def add(self, state: np.ndarray, action: np.ndarray, reward: float, 
            next_state: np.ndarray, done: bool) -> None:
        """Add a new experience to memory.
        
        Args:
            state: Current state observation
            action: Action taken
            reward: Reward received
            next_state: Next state observation
            done: Whether the episode ended
        """
        experience = self.experience(state, action, reward, next_state, done)
        self.memory.append(experience)

    def sample(self) -> Tuple[torch.Tensor, ...]:
        """Randomly sample a batch of experiences from memory.
        
        Returns:
            tuple: Batch of (states, actions, rewards, next_states, dones)
                  Each element is a torch tensor moved to the appropriate device
        """
        experiences = random.sample(self.memory, k=self.batch_size)
        
        # Filter out None experiences and convert to tensors
        valid_experiences = [e for e in experiences if e is not None]
        
        states = torch.from_numpy(
            np.vstack([e.state for e in valid_experiences])
        ).float().to(self.device)
        
        actions = torch.from_numpy(
            np.vstack([e.action for e in valid_experiences])
        ).float().to(self.device)
        
        rewards = torch.from_numpy(
            np.vstack([e.reward for e in valid_experiences])
        ).float().to(self.device)
        
        next_states = torch.from_numpy(
            np.vstack([e.next_state for e in valid_experiences])
        ).float().to(self.device)
        
        dones = torch.from_numpy(
            np.vstack([e.done for e in valid_experiences]).astype(np.uint8)
        ).float().to(self.device)

        return (states, actions, rewards, next_states, dones)

    def __len__(self) -> int:
        """Return the current size of internal memory."""
        return len(self.memory)

Actor Network Architecture¶

This class implements the Actor network component of the DDPG algorithm, which learns to predict the optimal action for any given state.

Network structure:

  • Input Layer: Takes state vectors as input
  • Hidden Layers: Two fully connected layers with 128 units each
  • Output Layer: Produces action vectors with tanh activation
  • Batch Normalization: Applied after first hidden layer for training stability

Key features:

  • ReLU activation functions in hidden layers
  • Tanh activation in output layer to bound actions between [-1, 1]
  • Batch normalization to prevent internal covariate shift
  • Configurable hidden layer sizes (default: 128 units)

The Actor network is essential in DDPG as it represents the policy π(s) that maps states to deterministic actions. The tanh activation ensures the output actions are properly scaled for the continuous control task.

import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Tuple

class ActorNetwork(nn.Module):
    """Actor (Policy) Network for DDPG algorithm.
    
    This network maps states to deterministic actions using a feed-forward neural network
    with batch normalization. The network architecture consists of three fully connected
    layers with ReLU activations and a final tanh activation to bound the actions.
    
    Attributes:
        fc1 (nn.Linear): First fully connected layer
        fc2 (nn.Linear): Second fully connected layer
        fc3 (nn.Linear): Output layer
        bn1 (nn.BatchNorm1d): Batch normalization after first layer
        
    Example:
        >>> actor = ActorNetwork(state_size=33, action_size=4, seed=0)
        >>> state = torch.randn(1, 33)
        >>> action = actor(state)  # Returns action values between -1 and 1
    """
    
    def __init__(self, 
                 state_size: int, 
                 action_size: int, 
                 seed: int, 
                 hidden_size1: int = 128, 
                 hidden_size2: int = 128) -> None:
        """Initialize the Actor Network.
        
        Args:
            state_size: Dimension of each state
            action_size: Dimension of each action
            seed: Random seed for reproducibility
            hidden_size1: Number of nodes in first hidden layer
            hidden_size2: Number of nodes in second hidden layer
        """
        super(ActorNetwork, self).__init__()
        
        # Set random seed for reproducibility
        self.seed = torch.manual_seed(seed)
        
        # Network layers
        self.fc1 = nn.Linear(state_size, hidden_size1)
        self.bn1 = nn.BatchNorm1d(hidden_size1)
        self.fc2 = nn.Linear(hidden_size1, hidden_size2)
        self.fc3 = nn.Linear(hidden_size2, action_size)
        
        # Initialize weights
        self._initialize_weights()
        
    def _initialize_weights(self) -> None:
        """Initialize network weights using Xavier initialization."""
        nn.init.xavier_uniform_(self.fc1.weight)
        nn.init.xavier_uniform_(self.fc2.weight)
        nn.init.xavier_uniform_(self.fc3.weight)

    def forward(self, state: torch.Tensor) -> torch.Tensor:
        """Build an actor (policy) network that maps states -> actions.
        
        Args:
            state: Current state tensor of shape (batch_size, state_size)
            
        Returns:
            torch.Tensor: Action values bounded between -1 and 1
        """
        # First hidden layer with batch normalization
        x = F.relu(self.fc1(state))
        x = self.bn1(x)
        
        # Second hidden layer
        x = F.relu(self.fc2(x))
        
        # Output layer with tanh activation to bound actions
        return F.tanh(self.fc3(x))

Critic Network Architecture¶

This class implements the Critic network component of the DDPG algorithm, which learns to estimate the Q-value (expected future rewards) for state-action pairs.

Network structure:

  • Input Processing:
    • State path: First processes the state through an initial hidden layer
    • Action path: Actions are concatenated with processed state features
  • Architecture:
    • First hidden layer (128 units) processes state only
    • Batch normalization for training stability
    • Second hidden layer (128 units) processes combined state-action features
    • Output layer produces single Q-value estimate

Key features:

  • ReLU activations in hidden layers
  • Batch normalization for stable training
  • State-action fusion through concatenation
  • Linear output for Q-value estimation

The Critic network is crucial for DDPG as it provides value estimates that guide the Actor's policy updates. Its architecture is specifically designed to effectively combine state and action information for accurate Q-value prediction.

import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Tuple

class CriticNetwork(nn.Module):
    """Critic (Value) Network for DDPG algorithm.
    
    This network estimates the Q-value for state-action pairs using a feed-forward neural network
    with batch normalization. The network architecture consists of three fully connected layers
    with ReLU activations.
    
    Attributes:
        fc1 (nn.Linear): First fully connected layer for state input
        fc2 (nn.Linear): Second fully connected layer for combined state-action input
        fc3 (nn.Linear): Output layer producing a single Q-value
        bn1 (nn.BatchNorm1d): Batch normalization after first layer
        
    Example:
        >>> critic = CriticNetwork(state_size=33, action_size=4, seed=0)
        >>> state = torch.randn(1, 33)
        >>> action = torch.randn(1, 4)
        >>> q_value = critic(state, action)  # Returns estimated Q-value
    """
    
    def __init__(self, 
                 state_size: int, 
                 action_size: int, 
                 seed: int, 
                 hidden_size1: int = 128, 
                 hidden_size2: int = 128) -> None:
        """Initialize the Critic Network.
        
        Args:
            state_size: Dimension of each state
            action_size: Dimension of each action
            seed: Random seed for reproducibility
            hidden_size1: Number of nodes in first hidden layer
            hidden_size2: Number of nodes in second hidden layer
        """
        super(CriticNetwork, self).__init__()
        
        # Set random seed for reproducibility
        self.seed = torch.manual_seed(seed)
        
        # Network layers
        self.fc1 = nn.Linear(state_size, hidden_size1)
        self.bn1 = nn.BatchNorm1d(hidden_size1)
        self.fc2 = nn.Linear(hidden_size1 + action_size, hidden_size2)
        self.fc3 = nn.Linear(hidden_size2, 1)
        
        # Initialize weights
        self._initialize_weights()
        
    def _initialize_weights(self) -> None:
        """Initialize network weights using Xavier initialization."""
        nn.init.xavier_uniform_(self.fc1.weight)
        nn.init.xavier_uniform_(self.fc2.weight)
        nn.init.xavier_uniform_(self.fc3.weight)

    def forward(self, state: torch.Tensor, action: torch.Tensor) -> torch.Tensor:
        """Build a critic (value) network that maps (state, action) pairs -> Q-values.
        
        Args:
            state: Current state tensor of shape (batch_size, state_size)
            action: Action tensor of shape (batch_size, action_size)
            
        Returns:
            torch.Tensor: Estimated Q-value for the given state-action pair
        """
        # First hidden layer with batch normalization
        x = F.relu(self.fc1(state))
        x = self.bn1(x)
        
        # Concatenate state and action, then pass through second hidden layer
        x = torch.cat([x, action], dim=1)
        x = F.relu(self.fc2(x))
        
        # Output layer producing Q-value
        return self.fc3(x)

Ornstein-Uhlenbeck Noise Implementation¶

This class implements the Ornstein-Uhlenbeck process for generating temporally correlated noise, which is crucial for exploration in continuous action spaces.

Key components:

  • Process Parameters:
    • mu: Mean of the noise process (default: 0.0)
    • theta: Rate of mean reversion (default: 0.1)
    • sigma: Scale of noise (default: 0.2)
    • sigma_decay: Decay rate for noise scale (default: 0.99)
    • sigma_min: Minimum noise scale (default: 0.1)

Main methods:

  • reset: Resets noise state and decays sigma
  • sample: Generates next noise sample using the OU process equation

Features:

  • Temporally correlated noise generation
  • Adaptive noise scaling through decay mechanism
  • Configurable parameters for exploration control

The OU noise process is particularly suitable for continuous control tasks as it provides smooth, temporally correlated exploration, which is more appropriate than random noise for physical systems.

"""
Reference: https://github.com/xkiwilabs/Multi-Agent-DDPG-using-PTtorch-and-ML-Agents/blob/master/OUNoise.py
"""

import numpy as np
import random
import copy
from typing import Any

class OUNoise:
    """Ornstein-Uhlenbeck process for generating noise in continuous action spaces.
    
    This process generates temporally correlated noise, which is useful for exploration
    in continuous action spaces. It is particularly suited for physical control problems
    where smooth exploration is desired.
    
    Attributes:
        mu (np.ndarray): Long-term mean of the noise process
        theta (float): Rate of mean reversion
        sigma (float): Scale of the noise
        sigma_min (float): Minimum value for sigma
        sigma_decay (float): Decay rate for sigma
        state (np.ndarray): Current state of the noise process
        
    Example:
        >>> noise = OUNoise(size=4, seed=0)
        >>> action = np.array([0.5, 0.1, -0.3, 0.0])
        >>> noisy_action = action + noise.sample()
    """
    
    def __init__(self, 
                 size: int, 
                 seed: int, 
                 mu: float = 0.0, 
                 theta: float = 0.1, 
                 sigma: float = 0.2, 
                 sigma_min: float = 0.1, 
                 sigma_decay: float = 0.99) -> None:
        """Initialize parameters and noise process.
        
        Args:
            size: Dimension of the noise
            seed: Random seed for reproducibility
            mu: Long-term mean of the noise process
            theta: Rate of mean reversion
            sigma: Initial scale of the noise
            sigma_min: Minimum value for sigma
            sigma_decay: Decay rate for sigma
        """
        self.mu = mu * np.ones(size)
        self.theta = theta
        self.sigma = sigma
        self.sigma_min = sigma_min
        self.sigma_decay = sigma_decay
        self.seed = random.seed(seed)
        self.size = size
        self.reset()

    def reset(self) -> None:
        """Reset the internal state to the mean (mu) and decay sigma."""
        self.state = copy.copy(self.mu)
        self.sigma = max(self.sigma_min, self.sigma * self.sigma_decay)

    def sample(self) -> np.ndarray:
        """Update internal state and return it as a noise sample.
        
        Returns:
            np.ndarray: Noise sample to be added to actions
        """
        x = self.state
        dx = self.theta * (self.mu - x) + self.sigma * np.random.standard_normal(self.size)
        self.state = x + dx
        return self.state

DDPG Agent Implementation¶

This class implements the Deep Deterministic Policy Gradient (DDPG) agent, combining actor-critic architecture with experience replay and noise-based exploration.

Deep Deterministic Policy Gradient (DDPG) Algorithm¶

DDPG is an off-policy algorithm that combines insights from DQN and deterministic policy gradients to handle continuous action spaces.

Core Components¶

1. Actor-Critic Architecture¶

  • Actor μ(s|θμ): Implements deterministic policy

    • Maps states to specific actions
    • Gradient: ∇θμJ ≈ Es~ρβ[∇aQ(s,a|θQ)|a=μ(s) ∇θμμ(s|θμ)]

  • Critic Q(s,a|θQ): Estimates action-value function

    • Updates using Bellman equation
    • Target: yi = ri + γQ'(si+1, μ'(si+1)|θQ')

2. Network Updates¶

  • Critic Update: Minimizes loss L = 1/N Σ(yi - Q(si,ai|θQ))2
  • Actor Update: Maximizes expected Q-value through policy gradient
  • Target Networks: Updated softly with parameter τ
    • θ' ← τθ + (1-τ)θ'

3. Exploration Strategy¶

  • Adds Ornstein-Uhlenbeck noise to deterministic actions
  • dxt = θ(μ-xt)dt + σdWt
    • θ: Mean reversion strength
    • μ: Long-term mean
    • σ: Noise scale

4. Experience Replay¶

  • Stores transitions (st, at, rt, st+1)
  • Breaks correlations in experience sequence
  • Enables mini-batch learning

Algorithm Flow¶

  1. Observe state st
  2. Execute action at = μ(st) + Nt
  3. Receive reward rt and new state st+1
  4. Store transition in replay buffer
  5. Sample random mini-batch
  6. Update critic by minimizing loss
  7. Update actor using policy gradient
  8. Soft update target networks

This architecture enables stable learning in continuous action spaces while maintaining sample efficiency through experience replay and exploration through noise injection.

Key Components:

  • Networks:

    • Actor (Local & Target): Determines optimal actions for states
    • Critic (Local & Target): Evaluates state-action pairs
    • Both use Adam optimizer with configured learning rate

  • Core Methods:

    • act: Selects actions using current policy and exploration noise
    • step: Stores experiences and triggers learning
    • learn: Updates both networks using sampled experiences
    • update_target_network: Performs soft updates of target networks

Features:

  • Separate target networks for stability
  • Experience replay buffer for sample efficiency
  • Ornstein-Uhlenbeck noise for exploration
  • Soft target updates with TAU parameter

import numpy as np
import torch
import torch.optim as optim
import torch.nn.functional as F
from typing import Any

class Agent:
    """DDPG Agent for interacting with and learning from the environment.
    
    This agent uses an actor-critic architecture with experience replay and
    Ornstein-Uhlenbeck noise for exploration in continuous action spaces.
    
    Attributes:
        actor_local (ActorNetwork): Local actor network
        actor_target (ActorNetwork): Target actor network
        actor_optimizer (optim.Optimizer): Optimizer for actor network
        critic_local (CriticNetwork): Local critic network
        critic_target (CriticNetwork): Target critic network
        critic_optimizer (optim.Optimizer): Optimizer for critic network
        memory (ReplayBuffer): Replay buffer for experience storage
        exploration_noise (OUNoise): Noise process for exploration
        
    Example:
        >>> agent = Agent(state_size=33, action_size=4, seed=0)
        >>> state = env.reset()
        >>> action = agent.act(state)
    """
    
    def __init__(self, state_size: int, action_size: int, seed: int) -> None:
        """Initialize an Agent object.
        
        Args:
            state_size: Dimension of each state
            action_size: Dimension of each action
            seed: Random seed for reproducibility
        """
        self.state_size = state_size
        self.action_size = action_size
        self.seed = random.seed(seed)

        # Actor networks
        self.actor_local = ActorNetwork(state_size, action_size, seed).to(device)
        self.actor_target = ActorNetwork(state_size, action_size, seed).to(device)
        self.actor_optimizer = optim.Adam(self.actor_local.parameters(), lr=LR)

        # Critic networks
        self.critic_local = CriticNetwork(state_size, action_size, seed).to(device)
        self.critic_target = CriticNetwork(state_size, action_size, seed).to(device)
        self.critic_optimizer = optim.Adam(self.critic_local.parameters(), lr=LR)

        # Replay memory
        self.memory = ReplayBuffer(action_size, BUFFER_SIZE, BATCH_SIZE, seed)

        # OUNoise
        self.exploration_noise = OUNoise(action_size, seed)

    def act(self, state: np.ndarray) -> np.ndarray:
        """Returns actions for given state as per current policy.
        
        Args:
            state: Current state observation
            
        Returns:
            np.ndarray: Action values clipped between -1 and 1
        """
        state = torch.from_numpy(state).float().unsqueeze(0).to(device)
        
        self.actor_local.eval()
        with torch.no_grad():
            action = self.actor_local(state).cpu().data.numpy()
        self.actor_local.train()

        # Add exploration noise
        action += self.exploration_noise.sample()

        return np.clip(action, -1, 1)

    def step(self, state: np.ndarray, action: np.ndarray, reward: float, 
             next_state: np.ndarray, done: bool) -> None:
        """Save experience in replay memory and use random sample from buffer to learn."""
        self.memory.add(state, action, reward, next_state, done)
        if len(self.memory) > BATCH_SIZE:
            self.learn()

    def learn(self) -> None:
        """Update policy and value parameters using batch of experience tuples."""
        states, actions, rewards, next_states, dones = self.memory.sample()
        next_actions = self.actor_target(next_states)
        y = rewards + GAMMA * self.critic_target(next_states, next_actions) * (1 - dones)
        
        # Critic loss
        critic_loss = F.mse_loss(y, self.critic_local(states, actions))
        
        # Critic backprop 
        self.critic_optimizer.zero_grad()
        critic_loss.backward()
        self.critic_optimizer.step()

        # Actor loss
        cur_actions = self.actor_local(states)
        actor_loss = -self.critic_local(states, cur_actions).mean()

        # Actor backprop
        self.actor_optimizer.zero_grad()
        actor_loss.backward()
        self.actor_optimizer.step()

        # Soft updates
        self.update_target_network()

    def update_target_network(self) -> None:
        """Soft update model parameters."""
        for target_param, local_param in zip(self.actor_target.parameters(), self.actor_local.parameters()):
            target_param.data.copy_(TAU * local_param.data + (1.0 - TAU) * target_param.data)

        for target_param, local_param in zip(self.critic_target.parameters(), self.critic_local.parameters()):
            target_param.data.copy_(TAU * local_param.data + (1.0 - TAU) * target_param.data)

import numpy as np
import torch
from collections import deque
from typing import List

def train(n_episodes: int = 1000, max_t: int = 1000) -> List[float]:
    """Train the DDPG agent in the environment.
    
    Args:
        n_episodes: Maximum number of training episodes
        max_t: Maximum number of timesteps per episode
        
    Returns:
        List[float]: Scores obtained in each episode
    """
    scores = []  # List to store scores from each episode
    scores_window = deque(maxlen=100)  # Last 100 scores for calculating average

    for i_episode in range(1, n_episodes + 1):
        agent.exploration_noise.reset()  # Reset noise for each episode
        
        env_info = env.reset(train_mode=True)[brain_name]  # Reset environment
        state = env_info.vector_observations[0]  # Get initial state
        score = 0  # Initialize score for the episode
        
        for _ in range(max_t):
            action = agent.act(state)  # Get action from agent
            env_info = env.step(action)[brain_name]  # Take action in environment
            next_state = env_info.vector_observations[0]  # Get next state
            reward = env_info.rewards[0]  # Get reward
            done = env_info.local_done[0]  # Check if episode is done
            
            agent.step(state, action, reward, next_state, done)  # Agent learns from experience
            state = next_state  # Move to next state
            score += reward  # Accumulate reward
            
            if done:
                break
        
        scores.append(score)  # Save most recent score
        scores_window.append(score)  # Save score to window
        mean_score = np.mean(scores_window)  # Calculate average score
        
        print(f"\rEpisode: {i_episode}, Average Score: {mean_score:.2f}", end="")
        
        # Save model if average score is above threshold
        if mean_score >= 30.0:
            torch.save(agent.actor_local.state_dict(), 'actor_checkpoint.pth')
            torch.save(agent.critic_local.state_dict(), 'critic_checkpoint.pth')
            print(f"\nEnvironment solved in {i_episode} episodes!")
            break
    
    return scores
